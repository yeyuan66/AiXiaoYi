  AgentState 状态定义 (agent/core.py:56-68)

  class AgentState(TypedDict):
      user_query: str              # 用户原始查询
      intent: IntentResult         # 意图识别结果
      task_list: List[Dict]        # 待执行任务列表
      current_task_idx: int        # 当前执行到第几个任务
      tool_results: List[Dict]     # 工具执行结果收集
      final_answer: str            # 最终答案
      memory: AgentMemory          # 记忆系统状态
      retry_count: int             # 重试次数
      step_count: int              # 已执行步数
      error: str                   # 错误信息

  🔗 节点间流转逻辑

  ┌──────────────────┬────────────────┬──────────────────────┐
  │     当前节点     │      条件      │       下一节点       │
  ├──────────────────┼────────────────┼──────────────────────┤
  │ intent_recognize │ 无条件         │ task_decompose       │
  ├──────────────────┼────────────────┼──────────────────────┤
  │ task_decompose   │ task_list非空  │ tool_execute         │
  ├──────────────────┼────────────────┼──────────────────────┤
  │ task_decompose   │ task_list为空  │ result_aggregate     │
  ├──────────────────┼────────────────┼──────────────────────┤
  │ tool_execute     │ 还有任务未完成 │ tool_execute（循环） │
  ├──────────────────┼────────────────┼──────────────────────┤
  │ tool_execute     │ 所有任务完成   │ result_aggregate     │
  ├──────────────────┼────────────────┼──────────────────────┤
  │ tool_execute     │ 出错           │ error_handle         │
  ├──────────────────┼────────────────┼──────────────────────┤
  │ error_handle     │ 重试次数未超限 │ tool_execute         │
  ├──────────────────┼────────────────┼──────────────────────┤
  │ error_handle     │ 超过最大重试   │ result_aggregate     │
  ├──────────────────┼────────────────┼──────────────────────┤

  🎯 状态流转检查函数 (agent/core.py)

  # 625-637行：检查是否有任务
  def _has_tasks(self, state: AgentState) -> str:
      task_list = state.get("task_list", [])
      return "has_tasks" if task_list else "no_tasks"

  # 638-666行：检查执行状态（决定是否继续或结束）
  def _check_execution_status(self, state: AgentState) -> str:
      # 1. 检查错误
      if state.get("error"):
          return "has_error"

      # 2. 检查执行步数限制
      step_count = state.get("step_count", 0)
      if step_count >= MAX_EXECUTE_STEPS:
          return "all_tasks_done"

      # 3. 检查是否还有任务
      task_list = state.get("task_list", [])
      current_idx = state.get("current_task_idx", 0)

      if current_idx >= len(task_list):
          return "all_tasks_done"
      else:
          return "has_more_tasks"

  🎯 任务拆解逻辑 (agent/core.py:241-301)

  def _decompose_tasks(self, query: str, intent: IntentResult) -> List[Dict]:
      """根据意图拆解为不同类型的任务"""

      if intent.intent == "knowledge_query":
          # 知识查询 → 调用RAG检索
          return [{
              "task_id": "task_0",
              "task_type": "rag_search",
              "params": {"query": query, "business_domain": "finance"}
          }]

      elif intent.intent == "data_query":
          # 数据查询 → 生成SQL
          return [{
              "task_id": "task_0",
              "task_type": "sql_execute",
              "params": {"query": query, "table_name": "product_info"}
          }]

      elif intent.intent == "task_operation":
          # 任务操作 → 简单Echo
          return [{"task_type": "echo", "params": {"message": query}}]

  🎯 重试机制 (agent/core.py:566-601)

  def _error_handle_node(self, state: AgentState) -> Dict[str, Any]:
      error = state.get("error", "未知错误")
      retry_count = state.get("retry_count", 0)

      if retry_count < MAX_RETRY_COUNT:  # 默认最多重试2次
          # 重试：清空错误，增加计数
          state["retry_count"] = retry_count + 1
          state["error"] = None
          return state

      # 超过重试次数 → 生成友好错误提示
      state["final_answer"] = self._generate_friendly_error(error)
      state["error"] = None
      return state

  def _execute_rag_search(self, task: Dict) -> Dict:
      # 构造MCP请求
      request = MCPRequestRequest(
          skill_name="rag_search",
          params=task.get("params", {}),
          trace_id=task.get("task_id")
      )
      # 通过调度器执行
      response = self._skill_scheduler.execute(request)
      return response.to_dict()
